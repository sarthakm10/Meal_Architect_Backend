-- 8) 7-DAY MEAL LOGS (from 2026-03-16 to 2026-03-22)
WITH plan(username, meal_a, meal_b) AS (
  VALUES
    ('john_doe',     'Chicken Rice Bowl',     'Egg Spinach Scramble'),
    ('maya_fit',     'Tofu Veggie Stir Fry',  'Oats Protein Bowl'),
    ('sarth_google', 'Lentil Rice Plate',     'Chicken Broccoli Prep')
),
days AS (
  SELECT (DATE '2026-03-22' - offs) AS consumed_date, offs
  FROM generate_series(0, 6) AS offs
)
INSERT INTO meal_logs (user_id, meal_id, consumed_date)
SELECT
  u.id,
  CASE WHEN d.offs % 2 = 0 THEN ma.id ELSE mb.id END AS meal_id,
  d.consumed_date
FROM plan p
JOIN users u ON u.username = p.username
JOIN meals ma ON ma.user_id = u.id AND ma.name = p.meal_a
JOIN meals mb ON mb.user_id = u.id AND mb.name = p.meal_b
CROSS JOIN days d
WHERE NOT EXISTS (
  SELECT 1
  FROM meal_logs ml
  WHERE ml.user_id = u.id
    AND ml.meal_id = CASE WHEN d.offs % 2 = 0 THEN ma.id ELSE mb.id END
    AND ml.consumed_date = d.consumed_date
);
